je_credit_currency = {
	should_be_pinned_by_default = yes
	group = je_group_internal_affairs
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	scripted_progress_bar = debt_amount_progress_bar
	scripted_progress_bar = broke_progress_bar

	is_shown_when_inactive = {
		always = yes
	}

	possible = {
		always = yes
	}

	immediate = {
		set_variable = {
			name = bank_reserves
			value = 0
		}
		set_variable = {
			name = total_debt
			value = 0
		}
		set_variable = {
			name = leverage_target
			value = 0
		}
		set_variable = {
			name = turnover_amount
			value = 0
		}
		set_variable = {
			name = this_season_gdp
			value = gdp
		}
		set_variable = {
			name = last_season_gdp
			value = gdp
		}
		set_variable = {
			name = confidence_factor
			value = 1
		}
		# for test
		set_variable = {
			name = reserves_rate
			value = 0.1
		}
		set_variable = {
			name = credit_level
			value = 0
		}
		set_variable = {
			name = bank_interest_rate_factor
			value = 1
		}
		set_variable = {
			name = cached_leverage_rate
			value = 0
		}
		set_variable = {
			name = cached_debt_bound
			value = 0
		}
		set_variable = {
			name = cached_risk_premium_factor
			value = 1
		}
		set_variable = {
			name = cached_trade_deficit_factor
			value = 1
		}
		set_variable = {
			name = cached_real_interest_rate
			value = 0.04
		}
	}

	on_weekly_pulse = {
		effect = {
			# Refresh economy-facing variables from bond production mix.
			set_variable = {
				name = bank_interest_rate_factor
				value = weighted_interest_factor
			}
			set_variable = {
				name = bank_reserves
				value = bank_reserves_from_bonds
			}
			set_variable = {
				name = turnover_amount
				value = turnover_from_bonds
			}
			set_variable = {
				name = cached_leverage_rate
				value = leverage_rate
			}
			set_variable = {
				name = cached_debt_bound
				value = debt_bound
			}
			set_variable = {
				name = cached_risk_premium_factor
				value = risk_premium_factor
			}
			set_variable = {
				name = cached_trade_deficit_factor
				value = trade_deficit_factor
			}
			set_variable = {
				name = cached_real_interest_rate
				value = real_interest_rate
			}

			change_variable = {
				name = total_debt
				add = weekly_interest
			}
			set_variable = {
				name = target_debt_change
				value = debt_change
			}
			recursive_loan_calculator = yes
			change_variable = {
				name = total_debt
				add = var:target_debt_change
			}
			add_investment_pool = var:target_debt_change
		}
	}
}
