recursive_loan_calculator = {
	if = {
		limit = { real_interest_rate < gdp_growth_rate }
	
		if = {
			limit ={
				target_interest_rate > gdp_growth_rate
				var:target_debt_change > 1000
			}
			change_variable = {
				name = target_debt_change
				divide = 2
			}
			recursive_loan_calculator = yes
		}
	}
	else = {
		if = {
			limit = {
				target_interest_rate < gdp_growth_rate
				var:target_debt_change < -1000
			}
			change_variable = {
				name = target_debt_change
				multiply = 0.5
			}
			recursive_loan_calculator = yes
		}
		
	}
	if = {
		limit = {
			target_debt > debt_bound 
		}
		set_variable = {
			name = target_debt_change
			value = {
				value = debt_bound
				subtract = var:total_debt
			}
		}
	}
}
